      html += `<button class="crumb ${isLast ? 'active' : ''}" data-idx="${i}">
        <i class="ph ph-${escapeAttr(t.icon || 'folder')}" style="color:${escapeAttr(t.iconColor || DEFAULT_COLOR)}"></i>${escape(t.name)}
      </button>`;
    });
    $('crumbs').innerHTML = html;
    $('crumbs').querySelectorAll('.crumb').forEach(b => {
      b.addEventListener('click', () => {
        const idx = parseInt(b.dataset.idx, 10);
        path = idx === -1 ? [] : path.slice(0, idx + 1);
        searchQ = ''; $('searchInput').value = '';
        render();
      });
    });
  }

  function renderContent() {
    const ce = $('content'); ce.innerHTML = '';

    const v = vaults[currentVaultId] || {};
    if (searchQ) {
      $('phTitle').innerHTML = `<i class="ph ph-magnifying-glass"></i>Search "${escape(searchQ)}"`;
    } else if (path.length === 0) {
      $('phTitle').innerHTML = `<i class="ph-bold ph-${escapeAttr(v.icon || 'vault')}" style="color:${escapeAttr(v.iconColor || DEFAULT_COLOR)}"></i>${escape(v.name || 'Vault')} · categories`;
    } else {
      const t = pathTitles().slice(-1)[0];
      $('phTitle').innerHTML = `<i class="ph-bold ph-${escapeAttr(t.icon || 'folder')}" style="color:${escapeAttr(t.iconColor || DEFAULT_COLOR)}"></i>${escape(t.name)}`;
    }

    // Pinned shelf at root
    if (!searchQ && path.length === 0) {
      const pinned = findAllPinned();
      if (pinned.length > 0) {
        const lbl = document.createElement('div');
        lbl.className = 'section-label';
        lbl.innerHTML = `<i class="ph-fill ph-push-pin"></i><span>Pinned</span><span style="color:var(--tx-faint);"> · ${pinned.length}</span>`;
        ce.appendChild(lbl);
        const strip = document.createElement('div');
        strip.className = 'grid pinned-strip';
        strip.innerHTML = pinned.map(p => cardHtml(p.id, p.item, p.fullPath)).join('');
        ce.appendChild(strip);
        attachCardEvents(strip);
      }
    }

    // Uploads
    if (!searchQ) {
      const ups = [...activeUploads.entries()]
        .filter(([, info]) => JSON.stringify(info.parentPath) === JSON.stringify(path));
      if (ups.length > 0) {
        const lbl = document.createElement('div');
        lbl.className = 'section-label';
        lbl.innerHTML = `<i class="ph ph-cloud-arrow-up"></i><span>Uploading · ${ups.length}</span>`;
        ce.appendChild(lbl);
        const u = document.createElement('div');
        u.className = view === 'grid' ? 'grid' : 'list';
        if (view === 'list') {
          u.innerHTML = ups.map(([tid, info]) => uploadListHtml(tid, info)).join('');
        } else {
          u.innerHTML = ups.map(([tid, info]) => uploadCardHtml(tid, info)).join('');
        }
        ce.appendChild(u);
      }
    }

    // Main entries
    let entries;
    if (searchQ) {
      entries = flatSearch(searchQ);
    } else {
      entries = Object.entries(currentItems()).map(([id, item]) => ({ id, item, fullPath: [...path, id] }));
    }
    entries = sortEntries(entries);
    $('phCount').textContent = `${entries.length} item${entries.length === 1 ? '' : 's'}`;

    if (entries.length === 0) {
      const empty = document.createElement('div');
      empty.className = 'empty';
      empty.innerHTML = `
        <i class="ph-thin ph-${path.length === 0 && !searchQ ? 'squares-four' : 'folder-open'}"></i>
        <h3>${searchQ ? 'Nothing found' : 'Nothing here yet'}</h3>
        <p>${searchQ ? 'Try a different term.' : (path.length === 0 ? 'Press + to add a category.' : 'Press + to add folders, URLs, files or notes.')}</p>
      `;
      ce.appendChild(empty);
      return;
    }

    const lbl = document.createElement('div');
    lbl.className = 'section-label';
    lbl.innerHTML = `<span>${searchQ ? 'Results' : 'All items'}</span><span style="color:var(--tx-faint);"> · ${entries.length}</span>`;
    ce.appendChild(lbl);

    if (view === 'grid') {
      const g = document.createElement('div');
      g.className = 'grid';
      g.innerHTML = entries.map(e => cardHtml(e.id, e.item, e.fullPath)).join('');
      ce.appendChild(g);
      attachCardEvents(g);
    } else {
      const wrap = document.createElement('div');
      wrap.className = 'list';
      wrap.innerHTML = `
        <div class="list-head">
          <div></div>
          <div>Name</div>
          <div class="col-type">Type</div>
          <div>Added</div>
          <div class="col-info">Info</div>
          <div></div>
        </div>
        ${entries.map(e => listRowHtml(e.id, e.item, e.fullPath)).join('')}
      `;
      ce.appendChild(wrap);
      attachCardEvents(wrap);
    }
  }

  function cardHtml(id, item, fullPath) {
    const type = item.type || (fullPath.length === 1 ? 'category' : 'folder');
    const ic = item.icon || DEFAULT_ICONS[type] || 'folder';
    const color = item.iconColor || DEFAULT_COLOR;
    let sub = '';
    if (type === 'url') {
      try { sub = new URL(item.url).hostname.replace(/^www\./, ''); } catch { sub = item.url || ''; }
    } else if (type === 'file') {
      sub = `${humanSize(item.size)} · ${item.mime || 'file'}`;
    } else if (type === 'note') {
      sub = (item.body || '').trim().slice(0, 200) || '(empty note)';
    } else {
      const c = item.items ? Object.keys(item.items).length : 0;
      sub = `${c} item${c === 1 ? '' : 's'}`;
    }
    const isImage = type === 'file' && (item.mime || '').startsWith('image/') && item.url;
    const pinned = !!item.pinned;
    const dataUrl = (type === 'url' || type === 'file') ? `data-url="${escapeAttr(item.url || '')}"` : '';

    if (isImage) {
      return `
        <div class="card thumb" data-type="${type}" data-name="${escapeAttr(item.name)}"
             data-mime="${escapeAttr(item.mime || '')}"
             data-fullpath='${escapeAttr(JSON.stringify(fullPath))}' ${dataUrl}>
          <div class="card-actions">
            <button class="ca pin ${pinned ? 'active' : ''}" title="Pin"><i class="${pinned ? 'ph-fill' : 'ph'} ph-push-pin"></i></button>
            <button class="ca color" title="Color"><i class="ph ph-palette"></i></button>
            <button class="ca dl" title="Download"><i class="ph ph-download-simple"></i></button>
            <button class="ca del" title="Delete"><i class="ph ph-trash"></i></button>
          </div>
          <div class="thumb-img" style="background-image:url('${escapeAttr(item.url)}')"></div>
          <div class="thumb-body">
            <div class="card-title">${escape(item.name)}${pinned ? '<span class="pinned-dot"></span>' : ''}</div>
            <div class="card-sub">${escape(humanSize(item.size))}</div>
          </div>
        </div>`;
    }

    return `
      <div class="card ${type}" data-type="${type}" data-name="${escapeAttr(item.name)}"
           data-mime="${escapeAttr(item.mime || '')}"
           data-fullpath='${escapeAttr(JSON.stringify(fullPath))}' ${dataUrl}>
        <div class="card-top">
          <div class="card-ic"><i class="ph-bold ph-${escapeAttr(ic)}" style="color:${escapeAttr(color)}"></i></div>
          <div class="card-actions">
            <button class="ca pin ${pinned ? 'active' : ''}" title="Pin"><i class="${pinned ? 'ph-fill' : 'ph'} ph-push-pin"></i></button>
            <button class="ca color" title="Color"><i class="ph ph-palette"></i></button>
            ${type === 'note' ? `<button class="ca edit-note" title="Edit"><i class="ph ph-pencil"></i></button>` : ''}
            ${type === 'url' ? `<button class="ca open" title="Open"><i class="ph ph-arrow-square-out"></i></button>` : ''}
            ${type === 'file' ? `<button class="ca dl" title="Download"><i class="ph ph-download-simple"></i></button>` : ''}
            <button class="ca del" title="Delete"><i class="ph ph-trash"></i></button>
          </div>
        </div>
        <div class="card-title">${escape(item.name)}${pinned ? '<span class="pinned-dot"></span>' : ''}</div>
        <div class="card-sub">${escape(sub)}</div>
      </div>
    `;
  }

  function listRowHtml(id, item, fullPath) {
    const type = item.type || (fullPath.length === 1 ? 'category' : 'folder');
    const ic = item.icon || DEFAULT_ICONS[type] || 'folder';
    const color = item.iconColor || DEFAULT_COLOR;
    const pinned = !!item.pinned;
    const dataUrl = (type === 'url' || type === 'file') ? `data-url="${escapeAttr(item.url || '')}"` : '';
    let info = '';
    if (type === 'url') {
      try { info = new URL(item.url).hostname.replace(/^www\./, ''); } catch { info = item.url || ''; }
    } else if (type === 'file') {
      info = humanSize(item.size);
    } else if (type === 'note') {
      const body = (item.body || '').trim();
      info = body ? `${body.length} chars` : '(empty)';
    } else {
      const c = item.items ? Object.keys(item.items).length : 0;
      info = `${c} item${c === 1 ? '' : 's'}`;
    }
    return `
      <div class="list-row" data-type="${type}" data-name="${escapeAttr(item.name)}"
           data-mime="${escapeAttr(item.mime || '')}"
           data-fullpath='${escapeAttr(JSON.stringify(fullPath))}' ${dataUrl}>
        <div class="card-ic"><i class="ph-bold ph-${escapeAttr(ic)}" style="color:${escapeAttr(color)}"></i></div>
        <div class="lr-name">${pinned ? '<i class="ph-fill ph-push-pin lr-pin"></i>' : ''}${escape(item.name)}</div>
        <div class="lr-type">${escape(type)}</div>
        <div class="lr-date">${escape(humanDate(item.updatedAt || item.createdAt))}</div>
        <div class="lr-info">${escape(info)}</div>
        <div class="lr-actions">
          <button class="ca pin ${pinned ? 'active' : ''}" title="Pin"><i class="${pinned ? 'ph-fill' : 'ph'} ph-push-pin"></i></button>
          <button class="ca color" title="Color"><i class="ph ph-palette"></i></button>
          ${type === 'note' ? `<button class="ca edit-note" title="Edit"><i class="ph ph-pencil"></i></button>` : ''}
          ${type === 'url' ? `<button class="ca open" title="Open"><i class="ph ph-arrow-square-out"></i></button>` : ''}
          ${type === 'file' ? `<button class="ca dl" title="Download"><i class="ph ph-download-simple"></i></button>` : ''}
          <button class="ca del" title="Delete"><i class="ph ph-trash"></i></button>
        </div>
      </div>`;
  }

  function attachCardEvents(scope) {
    scope.querySelectorAll('.card:not(.uploading), .list-row:not(.uploading)').forEach(el => {
      el.addEventListener('click', e => {
        if (e.target.closest('.ca')) return;
        const type = el.dataset.type;
        const fp = JSON.parse(el.dataset.fullpath);
        if (type === 'url') {
          window.open(el.dataset.url, '_blank', 'noopener');
        } else if (type === 'file') {
          const url = el.dataset.url;
          const mime = el.dataset.mime || '';
          if (mime.startsWith('image/')) openLightbox(url);
          else window.open(url, '_blank', 'noopener');
        } else if (type === 'note') {
          openNoteViewer(fp);
        } else {
          path = fp; searchQ = ''; $('searchInput').value = '';
          $('searchWrap').classList.remove('show');
          render();
          window.scrollTo({ top: 0, behavior: 'smooth' });
        }
      });
    });
    scope.querySelectorAll('.ca.del').forEach(b => {
      b.addEventListener('click', e => {
        e.stopPropagation();
        const row = b.closest('.card, .list-row');
        const fp = JSON.parse(row.dataset.fullpath);
        deleteItem(fp, row.dataset.name);
      });
    });
    scope.querySelectorAll('.ca.open').forEach(b => {
      b.addEventListener('click', e => {
        e.stopPropagation();
        const row = b.closest('.card, .list-row');
        if (row.dataset.type === 'url') window.open(row.dataset.url, '_blank', 'noopener');
      });
    });
    scope.querySelectorAll('.ca.dl').forEach(b => {
      b.addEventListener('click', e => {
        e.stopPropagation();
        const row = b.closest('.card, .list-row');
        const a = document.createElement('a');
        a.href = row.dataset.url; a.download = row.dataset.name; a.target = '_blank'; a.rel = 'noopener';
        document.body.appendChild(a); a.click(); a.remove();
      });
    });
    scope.querySelectorAll('.ca.pin').forEach(b => {
      b.addEventListener('click', e => {
        e.stopPropagation();
        const row = b.closest('.card, .list-row');
        const fp = JSON.parse(row.dataset.fullpath);
        togglePin(fp, b.classList.contains('active'));
      });
    });
    scope.querySelectorAll('.ca.color').forEach(b => {
      b.addEventListener('click', e => {
        e.stopPropagation();
        const row = b.closest('.card, .list-row');
        const fp = JSON.parse(row.dataset.fullpath);
        pickColorFor(fp);
      });
    });
    scope.querySelectorAll('.ca.edit-note').forEach(b => {
      b.addEventListener('click', e => {
        e.stopPropagation();
        const row = b.closest('.card, .list-row');
        const fp = JSON.parse(row.dataset.fullpath);
        openEditNote(fp);
      });
    });
  }

  function openNoteViewer(fullPath) {
    const node = nodeByFullPath(fullPath);
    if (!node) return;
    viewingNote = { fullPath, item: node };
    $('noteViewTitle').textContent = (node.name || 'NOTE').toUpperCase();
    const created = node.createdAt ? humanDateTime(node.createdAt) : '';
    const updated = node.updatedAt && node.updatedAt !== node.createdAt ? humanDateTime(node.updatedAt) : '';
    $('noteViewMeta').textContent = [
      created ? 'Created ' + created : '',
      updated ? 'Updated ' + updated : '',
      (node.body || '').length + ' chars'
    ].filter(Boolean).join('  ·  ');
    $('noteViewBody').textContent = node.body || '';
    $('noteBackdrop').classList.add('show');
  }

    $('noteBackdrop').classList.remove('show');
    viewingNote = null;

  function openEditNote(fullPath) {
    const node = nodeByFullPath(fullPath);

    selectedColor = node.iconColor || DEFAULT_COLOR;

    $('nameInput').value = node.name || '';
    $('urlInput').value = '';
    $('noteInput').value = node.body || '';
    $('iconSearch').value = '';

  // -------- Add modal --------
  function openAddModal(forceType) {
    if (!currentVaultId) return;
    editingNoteFullPath = null;
    const allowed = allowedTypes();
    selectedType = forceType && allowed.includes(forceType) ? forceType : allowed[0];
    selectedIcon = DEFAULT_ICONS[selectedType];
    selectedColor = DEFAULT_COLOR;
    pendingFile = null;
    $('filePreview').innerHTML = '';
    $('fileInput').value = '';
    $('nameInput').value = '';
    $('urlInput').value = '';
    $('noteInput').value = '';
    $('iconSearch').value = '';

    $('typePicker').querySelectorAll('.type-opt').forEach(b => {
      const t = b.dataset.type;
      b.disabled = !allowed.includes(t);
      b.classList.toggle('active', t === selectedType);
    });
    updateModalForType();

    if (path.length === 0) {
      $('modalSub').textContent = 'Categories sit at the top level.';
    } else {
      const titles = pathTitles();
      $('modalSub').textContent = `Adding to "${titles[titles.length - 1].name}"`;
    }
    renderSwatches();
    $('addBackdrop').classList.add('show');
    setTimeout(() => {
      if (selectedType === 'note') $('noteInput').focus();
      else $('nameInput').focus();
    }, 60);
  }
  function closeAdd() {
    $('addBackdrop').classList.remove('show');
    pendingFile = null;
    editingNoteFullPath = null;
  }

  function updateModalForType() {
    $('urlField').classList.toggle('hide', selectedType !== 'url');
    $('fileField').classList.toggle('hide', selectedType !== 'file');
    $('noteField').classList.toggle('hide', selectedType !== 'note');
    $('iconField').classList.toggle('hide', selectedType === 'file');
    $('nameLabel').textContent =
      selectedType === 'url' ? 'Label' :
      selectedType === 'file' ? 'Display name (optional)' :
      selectedType === 'note' ? 'Title' :
      'Name';
    $('nameInput').placeholder =
      selectedType === 'file' ? 'Auto-filled from file' :
      selectedType === 'note' ? 'Note title…' :
      'Give it a name…';
    if (!editingNoteFullPath) {
      $('modalTitle').textContent = `NEW ${selectedType.toUpperCase()}`;
    }
    if (!editingNoteFullPath || selectedType !== 'note') {
      selectedIcon = DEFAULT_ICONS[selectedType];
    }
    $('iconSearch').value = '';
    paintIconGrid($('iconGrid'), '', () => selectedIcon, (n) => { selectedIcon = n; updateIconPreview(); });
    updateIconPreview();
  }
  function updateIconPreview() {
    $('iconPreview').innerHTML = `<i class="ph-bold ph-${escapeAttr(selectedIcon)}" style="color:${escapeAttr(selectedColor)}"></i>`;
    $('colorPicker').value = selectedColor;

  }

  function handleFile(file) {
    pendingFile = file;
    if (!$('nameInput').value) $('nameInput').value = file.name;
    const isImg = file.type.startsWith('image/');
    let bg = '';
    if (isImg) {
      const url = URL.createObjectURL(file);
      bg = `style="background-image:url('${url}')"`;
    }
    $('filePreview').innerHTML = `
      <div class="file-preview">
        <div class="file-preview-thumb" ${bg}>
          ${!isImg ? `<i class="ph-bold ph-${iconForMime(file.type, file.name)}"></i>` : ''}
        </div>
        <div class="file-preview-info">
          <div class="file-preview-name">${escape(file.name)}</div>
          <div class="file-preview-size">${humanSize(file.size)} · ${escape(file.type || 'file')}</div>
        </div>
        <button class="ca" id="filePreviewClear"><i class="ph ph-x"></i></button>
      </div>`;
    $('filePreviewClear').addEventListener('click', () => {
      pendingFile = null;
      $('filePreview').innerHTML = '';
      $('fileInput').value = '';
    });
  }

    if ($('fileInput').files && $('fileInput').files[0]) handleFile($('fileInput').files[0]);
  });

  $('fileDrop').addEventListener('dragleave', () => $('fileDrop').classList.remove('dragover'));
  $('fileDrop').addEventListener('drop', e => {
    e.preventDefault(); $('fileDrop').classList.remove('dragover');
    if (e.dataTransfer.files && e.dataTransfer.files[0]) handleFile(e.dataTransfer.files[0]);
  });

  $('typePicker').querySelectorAll('.type-opt').forEach(b => {
    b.addEventListener('click', () => {

      selectedType = b.dataset.type;

      updateModalForType();

  });

  // -------- Save / CRUD --------
  async function saveItem() {
    // EDIT note
    if (editingNoteFullPath) {
      const name = $('nameInput').value.trim();
      const body = $('noteInput').value;
      if (!name) { toast('Title is required'); $('nameInput').focus(); return; }
      const node = nodeByFullPath(editingNoteFullPath) || {};
      const updated = {
        ...node,
        type: 'note',
        name,
        body,
        icon: selectedIcon,
        iconColor: selectedColor,
        updatedAt: Date.now(),
        createdAt: node.createdAt || Date.now()
      };
      try {
        await set(dbRef(db, fullPathRef(editingNoteFullPath)), updated);
        toast('Note saved');
        closeAdd();
      } catch (e) { console.error(e); toast('Failed'); }
      return;
    }

    if (selectedType === 'file') {
      if (!pendingFile) { toast('Pick a file first'); return; }
      const displayName = $('nameInput').value.trim() || pendingFile.name;
      const file = pendingFile;
      const parentPath = [...path];
      closeAdd();
      uploadFile(file, displayName, parentPath);
      return;
    }
    if (selectedType === 'note') {
      const name = $('nameInput').value.trim();
      const body = $('noteInput').value;
      if (!name && !body.trim()) { toast('Note needs a title or body'); $('nameInput').focus(); return; }
      const finalName = name || (body.trim().split('\n')[0].slice(0, 60) || 'Untitled note');
      const id = uid();
      const item = {
        type: 'note', name: finalName, body,
        icon: selectedIcon || 'note',
        iconColor: selectedColor,
        createdAt: Date.now(),
        updatedAt: Date.now()
      };
      try {
        await set(dbRef(db, nodeRefPath(id)), item);
        toast('Note saved');
        closeAdd();
      } catch (e) { console.error(e); toast('Failed'); }
      return;
    }

    const name = $('nameInput').value.trim();
    if (!name) { toast('Name is required'); $('nameInput').focus(); return; }
    let url = '';
    if (selectedType === 'url') {
      url = $('urlInput').value.trim();
      if (!url) { toast('URL is required'); $('urlInput').focus(); return; }
      if (!/^https?:\/\//i.test(url)) url = 'https://' + url;
    }
    const id = uid();
    const item = {
      type: selectedType, name,
      icon: selectedIcon,
      iconColor: selectedColor,
      createdAt: Date.now()
    };
    if (selectedType === 'url') item.url = url;
    try {
      await set(dbRef(db, nodeRefPath(id)), item);
      toast(`${selectedType[0].toUpperCase() + selectedType.slice(1)} added`);
      closeAdd();
    } catch (e) { console.error(e); toast('Failed'); }
  }

  async function uploadFile(file, displayName, parentPath) {
    if (!currentVaultId) return;
    const tempId = 'tmp_' + uid();
    activeUploads.set(tempId, { name: displayName, progress: 0, parentPath });
    if (JSON.stringify(parentPath) === JSON.stringify(path)) renderContent();

    const id = uid();
    const safeName = file.name.replace(/[^\w.\-]/g, '_');
    const storagePath = `vaults/${currentVaultId}/${id}_${safeName}`;
    const ref = stRef(storage, storagePath);
    const task = uploadBytesResumable(ref, file, { contentType: file.type || 'application/octet-stream' });

    task.on('state_changed',
      snap => {
        const pct = (snap.bytesTransferred / snap.totalBytes) * 100;
        const info = activeUploads.get(tempId);
        if (!info) return;
        info.progress = pct;
        if (JSON.stringify(parentPath) === JSON.stringify(path)) {
          const card = document.querySelector(`.uploading[data-temp="${tempId}"]`);
          if (card) {
            const bar = card.querySelector('.upbar > div');
            if (bar) bar.style.width = pct + '%';
            const sub = card.querySelector('.card-sub') || card.querySelector('.lr-info');
            if (sub) sub.textContent = view === 'grid' ? `${Math.round(pct)}% uploading` : `${Math.round(pct)}%`;
          } else { renderContent(); }
        }
      },
      err => {
        console.error(err); activeUploads.delete(tempId);
        if (JSON.stringify(parentPath) === JSON.stringify(path)) renderContent();
        toast('Upload failed');
      },
      async () => {
        try {
          const url = await getDownloadURL(task.snapshot.ref);
          const item = {
            type: 'file',
            name: displayName,
            icon: iconForMime(file.type, file.name),
            iconColor: DEFAULT_COLOR,
            url, storagePath,
            size: file.size, mime: file.type || '',
            createdAt: Date.now()
          };
          await set(dbRef(db, nodeRefPath(id, parentPath)), item);
          activeUploads.delete(tempId);
          toast('Uploaded');
        } catch (e) {
          console.error(e); activeUploads.delete(tempId);
          if (JSON.stringify(parentPath) === JSON.stringify(path)) renderContent();
          toast('Upload finished but save failed');
        }
      }
    );
  }

  async function deleteItem(fullPath, name) {
    if (!confirm(`Delete "${name}"? This can't be undone.`)) return;
    let node = currentCats()[fullPath[0]];
    for (let i = 1; i < fullPath.length; i++) {
      if (!node || !node.items) { node = null; break; }
      node = node.items[fullPath[i]];
    }
    const storagePaths = [];
    (function collect(n) {
      if (!n) return;
      if (n.type === 'file' && n.storagePath) storagePaths.push(n.storagePath);
      if (n.items) for (const c of Object.values(n.items)) collect(c);
    })(node);
    try {
      await remove(dbRef(db, fullPathRef(fullPath)));
      for (const sp of storagePaths) {
        try { await deleteObject(stRef(storage, sp)); } catch (e) { console.warn(e); }
      }
      toast(storagePaths.length ? `Deleted (+${storagePaths.length} file${storagePaths.length === 1 ? '' : 's'})` : 'Deleted');
    } catch (e) { console.error(e); toast('Failed'); }
  }

  async function togglePin(fullPath, isPinned) {

    } catch (e) { console.error(e); toast('Failed'); }
  }

  }

  async function saveVault() {
    const name = $('vaultName').value.trim();
    if (!name) { toast('Name is required'); $('vaultName').focus(); return; }
    const id = uid();
    try {
      await set(dbRef(db, `${ROOT}/vaults/${id}`), {
        name, icon: editingVaultIcon, iconColor: editingVaultColor, createdAt: Date.now()
      });
      currentVaultId = id; path = [];
      toast('Vault created');
      closeVaultModal();
    } catch (e) { console.error(e); toast('Failed'); }
  }
  async function deleteVault(id) {
    const v = vaults[id]; if (!v) return;
    if (Object.keys(vaults).length <= 1) { toast('Need at least one vault'); return; }
    if (!confirm(`Delete vault "${v.name}" and everything inside?`)) return;
    const sps = [];
    (function walk(items) {
      for (const it of Object.values(items || {})) {
        if (it.type === 'file' && it.storagePath) sps.push(it.storagePath);
        if (it.items) walk(it.items);
      }
    })(allData[id] || {});
    try {
      await remove(dbRef(db, `${ROOT}/vaults/${id}`));
      await remove(dbRef(db, `${ROOT}/data/${id}`));
      for (const sp of sps) { try { await deleteObject(stRef(storage, sp)); } catch (e) { console.warn(e); } }
      if (currentVaultId === id) { currentVaultId = null; path = []; }
      toast(sps.length ? `Vault deleted (+${sps.length} files)` : 'Vault deleted');
    } catch (e) { console.error(e); toast('Failed'); }
  }
